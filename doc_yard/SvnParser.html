def get_root_directories
  dirs      = []
  rootindex = Browser.get(@svn_root).body

  rootindex.scan(%r{<li><a href=".+">(.+)/</a></li>}i).each do |dir|
    dirs << dir[0]
  end

  dirs.sort!
  dirs.uniq
end
